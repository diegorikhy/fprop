HTML structure

Dhromatron template is a fixed or fluid layout container with two nested columns. Left sided column represent main navigation block with .navigation-block class. All the content goes to the right column with .content-block class. General template structure can be found here.

The best way how to get familiar with each element is to take a look at code itself. Every party is well documented and with combination with origianl Bootstrap documentation you should not get lost. But if it happens and you donâ€™t know what to do feel free to contact us.

<body>

	<!-- Main page container -->
	<div class="container" role="main">

		<!-- Left (navigation block) side -->
		<div class="navigation-block">

			<!-- Navigation block widgets -->
			<section class="user-profile">
				...
			</section>

			<form class="side-search">
				...
			</form>

			<!-- Responsive navigation button -->
			<div class="navbar">
				<a href="#" class="btn btn-navbar"> ... </a>
			</div>

			<!-- Main navigation -->
			<nav class="main-navigation" role="navigation">
				...
			</nav>

		</div>
		<!-- /Left (navigation block) side -->

		<!-- Right (content block) side -->
		<div class="content-block">

			<!-- Widget container -->
			<div class="widget-container">
				...
			</div>

			<!-- Breadcrumbs -->
			<ul class="breadcrumbs">
				<li><a href="#"> ... </a></li>
				<li class="active"><a href="#"> ... </a></li>
			</ul>

			<!-- Page header -->
			<article class="page-header">
				...
			</article>

			<!-- Grid row -->
			<div class="row">

				<!-- Data block -->
				<article class="spanX data-block">
					<div class="data-container">
						...
					</div>
				</article>

			</div>

		</div>
		<!-- /Right (content block) side -->

	</div>
	<!-- /Main page container -->

</body>
